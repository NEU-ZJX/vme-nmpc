79 void get_token( FILE *fd, char *buffer, char *lastdelim, int *lineno )
80  {
81  char ch = '\0';
82  char *eob = buffer + BUFSIZ - 1;
83  char *curb = buffer;
84 
85  // Read the file one character at a time while not at end of file...
86  //
87  while ((ch = fgetc(fd)) != EOF)
88  {
89  // If we hit white space, gobble it, and keep track of new lines:
90  //
91  if ( ch == ' ' || ch == '\t' || ch == '\n' )
92  {
93  if ( ch == '\n' ) ++(*lineno);
94  continue;
95  }
96  //
97  // Otherwise, if we hit a '#', gobble the rest of the line, and count the
98  // new line:
99  //
100  else if ( ch == '#' )
101  {
102  while (ch != '\n')
103  ch = fgetc(fd);
104  ++(*lineno);
105  }
106  //
107  // If we see a delimeter, we shouldn't expect one!
108  //
109  else if ( ch == '=' || ch == ':' )
110  {
111  sprintf(buffer, "Line: %d, Character: '%c'", *lineno, ch);
112  report_error(RECOVERABLE_INPUT_FILE_SYNTAX, buffer);
113  buffer[0] = '\0';
114  return;
115  }
116  //
117  // Made it through ignormable (new word?) stuff, so start recording the
118  // characters into the buffer:
119  //
120  else
121  {
122  *curb++ = ch;
123  //
124  // Any of these characters ends the recording:
125  //
126  while ((ch = fgetc(fd)) != EOF && ch != ' ' && ch != '\t'
127  && ch != '=' && ch != ',' && ch != ':' && ch != '\r'
128  && ch != ';' && ch != '\n')
129  *curb++ = ch;
130  //
131  // We should expect that the recording was ended by these characters,
132  // otherwise, throw an error.
133  //
134  if ( ch != ' ' && ch != ',' && ch != ';' && ch != '\t' && ch != '='
135  && ch != '\n' )
136  {
137  // This error should really never happen unless we get \r.
138  sprintf(buffer,
139  "Line %d: Found invalid character '\\r' ending a token.",
140  *lineno, ch);
141  report_error(INVALID_INPUT_FILE_SYNTAX, buffer);
142  }
143  else
144  {
145  //
146  // By this point, a token has been recorded, and we have an
147  // acceptible termination. Check to see if it was terminated with
148  // a new line.
149  //
150  if ( ch == '\n' ) lineno++;
151  //
152  // If there is white space between the end of the token and its
153  // delimeter, then gobble it.
154  //
155  while (ch == ' ' || ch == '\n' || ch == '\t' || ch == '\r')
156  {
157  if ( ch == '\n' ) ++(*lineno);
158  ch = fgetc(fd);
159  }
160  //
161  // Regardless of wheather or not the token was ended with white
162  // space, or an accpeted delimeter, ch should hold the delimeter
163  // at this point. Record it.
164  //
165  if ( ch == '=' || ch == ',' || ch == ';' ) *lastdelim = ch;
166  //
167  // Accept ':' as the same as '='. This behaviour may change later,
168  // if I want '=' to be used for adjustable parameters, and ':' if
169  // the code should keep them constant.
170  //
171  if ( ch == ':' ) *lastdelim = '=';
172  }
173  break;
174  }
175  }
176  //
177  // Terminate the string in the buffer:
178  //
179  *curb = '\0';
180  //
181  // If we staid out of all of these decision trees, then ch could (should)
182  // be holding the EOF character for the file. Record it to let the calling
183  // routine know that we've hit the end of the file.
184  //
185  if ( ch == EOF)
186  {
187  *lastdelim = EOF;
188  //
189  // If the input file was made by a descent editor, there will be a
190  // newline character at the end of the file, which shouldn't count towards
191  // the line count. Bump it down.
192  //
193  --(*lineno);
194  }
195  }

215 void parse_input_file( nmpc &controller, const char *infile )
216  {
217  int k = 0, lineno = 1;
218  char *token, lastdelim;
219  char buffer[BUFSIZ];
220  char errmsg[256];
221  FILE *infd;
222 
223  infd = fopen(infile, "r");
224  if ( infd == NULL ) report_error(CANNOT_OPEN_INFILE, NULL);
225  //
226  // Loop, reading each record from the input file.
227  //
228  while (1)
229  {
230  // Begin reading a record. The first token out should be the key that
231  // identifies which variable is being set. E.g. the 'T' in "T = 0.3;".
232  //
233  get_token(infd, buffer, &lastdelim, &lineno);
234  if ( lastdelim == EOF) break;
235  //
236  // The delim of the first token should be '=', otherwise something has
237  // gone wrong:
238  //
239  if ( lastdelim != '=' && lastdelim != EOF)
240  {
241  sprintf(errmsg,
242  "Line %d: Invalid field delimeter '%c', expected '=' or ':'.",
243  lineno, lastdelim);
244  report_error(INVALID_INPUT_FILE_SYNTAX, errmsg);
245  }
246  //
247  // Now that we should have the key, try to identify it and record the
248  // following tokens in their appropriate memory locations.
249  // With error checking!
250  //
251  if ( strcmp("T", buffer) == 0 )
252  {
253  get_token(infd, buffer, &lastdelim, &lineno);
254  check_delim(lastdelim, ';', lineno);
255  controller.T = atof(buffer);
256  }
257  else if ( strcmp("N", buffer) == 0 )
258  {
259  get_token(infd, buffer, &lastdelim, &lineno);
260  check_delim(lastdelim, ';', lineno);
261  controller.N = atoi(buffer);
262  }
263  else if ( strcmp("C", buffer) == 0 )
264  {
265  get_token(infd, buffer, &lastdelim, &lineno);
266  check_delim(lastdelim, ';', lineno);
267  controller.C = atoi(buffer);
268  }
269  else if ( strcmp("m", buffer) == 0 )
270  {
271  get_token(infd, buffer, &lastdelim, &lineno);
272  check_delim(lastdelim, ';', lineno);
273  controller.m = atoi(buffer);
274  }
275  else if ( strcmp("n", buffer) == 0 )
276  {
277  get_token(infd, buffer, &lastdelim, &lineno);
278  check_delim(lastdelim, ';', lineno);
279  controller.n = atoi(buffer);
280  }
281  else if ( strcmp("dg", buffer) == 0 )
282  {
283  get_token(infd, buffer, &lastdelim, &lineno);
284  check_delim(lastdelim, ';', lineno);
285  controller.dg = atof(buffer);
286  }
287  else if ( strcmp("cruising_speed", buffer) == 0 )
288  {
289  get_token(infd, buffer, &lastdelim, &lineno);

292  }
293  else if ( strcmp("eps", buffer) == 0 )

295  get_token(infd, buffer, &lastdelim, &lineno);

298  }

300  {
301  controller.R = new float[2];
302  for ( k = 0; k < controller.n; ++k )
303  {
304  get_token(infd, buffer, &lastdelim, &lineno);
305  if ( k == controller.n - 1 )
306  {
307  check_delim(lastdelim, ';', lineno);
308  }
309  else
310  {
311  check_delim(lastdelim, ',', lineno);
312  }
313  controller.R[k] = atof(buffer);
314  }
315  }
316  else if ( strcmp("Q0", buffer) == 0 )
317  {
318  controller.Q0 = new float[2];
319  for ( k = 0; k < 2; ++k )
320  {
321  get_token(infd, buffer, &lastdelim, &lineno);
322  if ( k == 1 )
323  {
324  check_delim(lastdelim, ';', lineno);
325  }
326  else
327  {
328  check_delim(lastdelim, ',', lineno);
329  }
330  controller.Q0[k] = atof(buffer);
331  }
332  }
333  else if ( strcmp("Q", buffer) == 0 )
334  {
335  controller.Q = new float[2];
336  for ( k = 0; k < 2; ++k )
337  {
338  get_token(infd, buffer, &lastdelim, &lineno);
339  if ( k == 1 )
340  {
341  check_delim(lastdelim, ';', lineno);
342  }
343  else
344  {
345  check_delim(lastdelim, ',', lineno);
346  }
347  controller.Q[k] = atof(buffer);
348  }
349  }

352  controller.S = new float[controller.m];
353  for ( k = 0; k < controller.m; ++k )
354  {
355  get_token(infd, buffer, &lastdelim, &lineno);

358  check_delim(lastdelim, ';', lineno);

366  }

384  }

391  while (lastdelim != ';')
392  get_token(infd, buffer, &lastdelim, &lineno);

394  }
