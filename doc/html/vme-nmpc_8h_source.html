37 inline float costfun( const qnu *qu, const Lagr *p, const nmpc &C )
38  {
39  float J; // The cost
40  J = C.Q0[0] * p[C.N - 1].ex * p[C.N - 1].ex
41  + C.Q0[1] * p[C.N - 1].ey * p[C.N - 1].ey;
42  for ( unsigned int k = 0; k < C.N-1; k++ )
43  {
44  J += 2 * p[C.N - 1].ex * p[C.N - 1].ex * C.Q0[0]
45  + 2 * p[C.N - 1].ey * p[C.N - 1].ey * C.Q0[1];
46  for ( unsigned int j = 0; j < C.obst->size() / 2; j++ )
47  J += 1
48  / (((*C.tgt)[0] - qu[0].x) * ((*C.tgt)[0] - qu[0].x)
49  + ((*C.tgt)[1] - qu[0].y) * ((*C.tgt)[1] - qu[0].y)
50  + C.eps);
51  }
52  return -J;
53  }

61 inline void predict_horizon( qnu *qu, Lagr *p, const nmpc &C )
62  {
63  float dirx = (*C.tgt)[0] - qu[0].x;
64  float diry = (*C.tgt)[1] - qu[0].y;
65  // TODO: Store dist this to use in loop terminator.
66  float dist = sqrt(dirx * dirx + diry * diry);
67  dirx /= dist;
68  diry /= dist;
69  /*
70  * For all times in the horizon, provide an approximation of the state
71  * vector, based on a simple Euler integration scheme.
72  */
73  for ( unsigned int k = 1; k < C.N; k++ )
74  {
75  qu[k].th = qu[k - 1].th + qu[k - 1].Dth * C.T; // Steering angle.
76  p[k].costk = cos(qu[k].th);
77  p[k].sintk = sin(qu[k].th);
78  qu[k].x = qu[k - 1].x + qu[k - 1].Dx * C.T; // X-position.
79  qu[k].Dx = C.cruising_speed * p[k].costk; // X-speed.
80  qu[k].y = qu[k - 1].y + qu[k - 1].Dy * C.T; // Y-position.
81  qu[k].Dy = C.cruising_speed * p[k].sintk; // Y-speed.
82  p[k].ex = qu[k].x - (qu[0].x + C.cruising_speed * dirx * k * C.T);
83  p[k].ey = qu[k].y - (qu[0].y + C.cruising_speed * diry * k * C.T);
84  }
85  }

90 inline void get_gradient( qnu *qu, Lagr *p, nmpc &C, float *grad )
91  {
92  int k;
93  unsigned int j;
94  float PhiX, PhiY, denom, difx, dify, gDth;
95 
96  grad[C.N] = 0.;
97 
98  p[C.N - 1].p1 = C.Q0[0] * p[C.N - 1].ex;
99  p[C.N - 1].p3 = C.Q0[1] * p[C.N - 1].ey;
100  qu[C.N - 2].Dth -= C.dg * C.R[0] * qu[C.N - 2].Dth;
107  printf("# k p1 p2 p3 p4 p5");
108  printf(" PhiX PhiY grad[k]\n");
109  printf("# 19% 10.4f% 10.4f% 10.4f% 10.4f% 10.4f\n", p[C.N - 1].p1,
110  p[C.N - 1].p2, p[C.N - 1].p3, p[C.N - 1].p4, p[C.N - 1].p5);
111  for ( k = C.N - 2; k >= 0; --k )
112  {
113  PhiX = 0.;
114  PhiY = 0.;
115  /*
116  * Compute the obstacle potential by looping through the list of
117  * obstacles:
118  */
119  for ( j = 0; j < C.obst->size() / 2; j++ )
120  {
121  /*
122  * FIXME: Probably best to remove the vectors from C and use
123  * arrays. Vectors could be used temporarily to expand the
124  * data from the input file to memory. Error checking for
125  * the number of coordinates can be done there.
126  */
127  difx = qu[k].x - (*C.obst)[j * 2];
128  dify = qu[k].y - (*C.obst)[(j * 2) + 1];
129  denom = difx * difx + dify * dify + C.eps;
130  denom *= denom;
131  PhiX += 2 * difx / denom;
132  PhiY += 2 * dify / denom;
133  }
134  // Compute the Lagrange multipliers:
135  p[k].p1 = C.Q[0] * p[k].ex - PhiX + p[k + 1].p1;
136  p[k].p2 = p[k + 1].p1 * C.T;
137  p[k].p3 = C.Q[1] * p[k].ey - PhiY + p[k + 1].p3;
138  p[k].p4 = p[k + 1].p3 * C.T;
139  p[k].p5 = p[k + 1].p5 + p[k + 1].p4 * qu[k].v * p[k].costk
140  - p[k + 1].p2 * qu[k].v * p[k].sintk;
141  grad[k] = C.R[0] * qu[k].Dth + p[k + 1].p5 * C.T;
142  grad[C.N] += grad[k] * grad[k];
143  printf("# %2d% 10.4f% 10.4f% 10.4f% 10.4f% 10.4f", k, p[k].p1, p[k].p2,
144  p[k].p3, p[k].p4, p[k].p5);
145  printf("% 10.4f% 10.4f% 10.4f\n", PhiX, PhiY, grad[k]);
146  }
147  grad[C.N] = sqrt(grad[C.N]);
148  }
